\section*{Importance Sampling}
\begin{description}
	\item[Task:] Compute the integral
	\begin{equation}
		I = \frac{1}{\pi}\int_{-\infty}^{+\infty}\exp(-y_1^2-y_2^2)\,\mathrm{d}y_1\mathrm{d}y_2
	\end{equation}
	numerically with Monte Carlo methods; compare equally distributed random numbers (RN's) again with an appropriate importance sampling. \textit{Hints:}
	\begin{itemize}
		\item First use equally distributed RN's in $[-5,\,+5]$.
		\item Second use importance sampling with a suitable function. \textit{Hint:} Use the function
		provided in the Box Muller algorithm as given in the lecture:
		\begin{equation}
			y_1 = \sqrt{-2\ln(x_1)}\,\cos(2\pi x_2)
			\qquad
			y_2 = \sqrt{-2\ln(x_1)}\,\sin(2\pi x_2)
			\qquad\text{with $x_1,\,x_2\in [0,\,1]$}
		\end{equation}
	\end{itemize}
\end{description}


\section*{Random Walk}
\begin{description}
	\item[Task:] Construct a stochastic process, which has the probability density function
	\begin{equation}
		g(y_1,\,y_2) = \frac{1}{2\pi}\exp\Bigl\{-\frac{y_1^2}{2}-\frac{y_2^2}{2}\Bigr\}
	\end{equation}
	as an equilibrium distribution (Metropolis method). After an initial phase of $i$ random walk steps store the sequence of $k$ pairs of numbers $y_1,\,y_2$; the sequence $y_1,\,y_2$ is a stochastic representation of the underlying equilibrium distribution. 
	For each sequence a ``measurement'' of the value of the integral is obtained, and the expectation value is the average of many such measurements.\\
	Create $N\in\{100,\,1000,\,10000\}$ of such sequences $y_1,\,y_2$ and determine the integral as average of all measurements, and plot the error as function of $N$. What is a good choice for the number of initial steps $i$? Just check for convergence of the result.
\end{description}
